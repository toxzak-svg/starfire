# Training loop
model.train()
loss_fn = nn.CrossEntropyLoss(ignore_index=-1)

for step, (inp, tgt) in enumerate(dataloader):
    if step >= STEPS: break
    inp, tgt = inp.cuda(), tgt.cuda()
    
    # Forward pass
    outputs = model(inp)
    logits = outputs['logits']
    
    # Compute loss
    loss = loss_fn(logits.view(-1, logits.size(-1)), tgt.view(-1))
    
    # Backward
    optimizer.zero_grad()
    loss.backward()
    torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
    optimizer.step()
    
    if step % 50 == 0:
        with torch.no_grad():
            outputs = model(inp[:8])
            g = torch.stack(outputs['gates']).mean(dim=(0,1)).cpu().numpy()
        print(f"Step {step}/{STEPS} | Loss: {loss.item():.4f} | Gates: c={g[0]:.2f} s={g[1]:.2f} a={g[2]:.2f}")
        wandb.log({'step': step, 'loss': loss.item(), 'gate_cheap': g[0], 'gate_ssm': g[1], 'gate_attn': g[2]})

print("Done!")
wandb.log({'final_loss': loss.item()})
wandb.finish()

# Save
torch.save(model.module.state_dict() if isinstance(model, nn.DataParallel) else model.state_dict(), 
           '/kaggle/working/neu-optimal.pt')
print("Saved to /kaggle/working/neu-optimal.pt")
